37/* wakeup pin 3, 4 only available in 8x8 */
38void rtc_wakeup_sample_ext_cb(void)
39{
40 unsigned long long time_old;
41 UINT32 ioctl = 0;
42
43 time_old = RTC_GET_COUNTER();
44 /* Waits until the RTC register is updated*/
45 while (RTC_GET_COUNTER() < (time_old + 10));
46 RTC_IOCTL(RTC_GET_GPIO_WAKEUP_CONTROL_REG, &ioctl);
47
48 if (ioctl & RTC_WAKEUP_STATUS) {
49 Printf(">>> rtc1 wakeup interrupt ...\r\n" );
50 }
51 if (ioctl & RTC_WAKEUP2_STATUS) {
52 Printf(">>> rtc2 wakeup interrupt ...\r\n" );
53 }
54 if (ioctl & RTC_WAKEUP3_STATUS) {
55 Printf(">>> rtc3 wakeup interrupt ...\r\n" );
56 }
57 if (ioctl & RTC_WAKEUP4_STATUS) {
58 Printf(">>> rtc4 wakeup interrupt ...\r\n" );
59 }
60
61 /* clear wakeup source */
62 time_old = RTC_GET_COUNTER();
63 RTC_CLEAR_EXT_SIGNAL();
64 while (RTC_GET_COUNTER() < (time_old + 1));
65
66 time_old = RTC_GET_COUNTER();
67 ioctl = 0;
68 RTC_IOCTL(RTC_SET_WAKEUP_SOURCE_REG, &ioctl);
69 while (RTC_GET_COUNTER() < (time_old + 1));
70}
