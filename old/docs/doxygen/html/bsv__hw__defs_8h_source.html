26 /* poll NVM register to assure that the NVM boot is finished (and LCS and the keys are valid) */
27 #define CC_BSV_WAIT_ON_NVM_IDLE_BIT(hwBaseAddress) \
28  do { \
29  uint32_t regVal; \
30  do { \
31  SB_HAL_READ_REGISTER( SB_REG_ADDR(hwBaseAddress, NVM_IS_IDLE), regVal); \

33  }while( !regVal ); \
34  }while(0)

114 /* calc OTP memory length:
115  read RTL OTP address width. The supported sizes are 6 (for 2 Kbits),7,8,9,10,11 (for 64 Kbits).
116  convert value parameter to addresses of 32b words */
117 #define CC_BSV_GET_OTP_LENGTH(hwBaseAddress, otpLength) \
118  do { \
119  SB_HAL_READ_REGISTER(SB_REG_ADDR(hwBaseAddress, OTP_ADDR_WIDTH_DEF), otpLength); \
120  otpLength = CC_REG_FLD_GET(0, OTP_ADDR_WIDTH_DEF, VALUE, otpLength); \
121  otpLength = (1 << otpLength); \
122  }while(0)
